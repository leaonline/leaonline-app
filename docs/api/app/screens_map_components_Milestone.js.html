const getStars = (value) => {
  const stars = []
  stars.length = value

  const xOffsetLeft = ((value - 1) * 9) / 2

  for (let i = 0; i < value; i++) {
    stars[i] = (<MilestoneStar key={i} index={i} xOffsetLeft={xOffsetLeft} />)
  }

  return stars
}

const MilestoneStar = React.memo((props) => {
